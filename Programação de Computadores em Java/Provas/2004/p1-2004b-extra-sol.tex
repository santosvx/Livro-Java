class Q1 {
  public static void main (String[] a) {
    int n;
    do { n = lêInt();
         if (n<=0) break;
         imprime(somat(n));
       }
    while (true);
  }

  public static int somat(int n) {
    int s = 0, sinal = 1, num = 1, denom = 1;
    for (int i=0; i<=n; i++) {
      s = s + sinal * (num/(float)denom); }
      sinal = -sinal; num++; denom+=2*n-1;
    }
  }
}

class Q3 {
  public static void main (String[] a) {
    int n;
    do { n = lêInt();
         if (n<0) break;
         imprime(binDec(decBin(n)));
       }
    while (true);
  }

  public static String decBin(int n) {
    if (n==0) return "";
    else return decBin(n/2) + n%2;
  }

  public static int binDec(String s) {
    return binDec(s, 0, 0); }

  private static int binDec(String s, int i, int n) {
    if (i==s.length) return n;
    else return binDec(s,i+1,n*2+s.charAt(i)-'0');
    }
  }
}
